{{/* string-to-array.html Converts comma and colon delimited string-arrays to arrays Spaces before and after strings will be removed @author @sean-au @context string @return input with converstion complete @access private @example - Go Template {{ partial "hri/private/utils/string-to-array" "1,2,3" }} {{ partial "hri/private/utils/string-to-array" "1, 2, 3" }} */}} {{ $param := . }} {{/* check for comma delimited array (string) */}} {{ if (findRE "," $param) }} {{/* create array ',' delimiter for creation */}} {{ $param = split . "," -}} {{/* check for colon delimited array (string) */}} {{ else if (findRE ":" $param) }} {{/* create array ':' delimiter for creation */}} {{ $param = split . "," -}} {{ end }} {{/* remove leading spaces from each string */}} {{ $param = apply $param "strings.TrimLeft" " " "." }} {{/* remove trailing spaces from each string */}} {{ $param = apply $param "strings.TrimRight" " " "." }} {{ return $param }}
